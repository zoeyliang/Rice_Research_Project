% ====================================================
\section{Modified iterative sample size estimation for MFMC}\label{sec:Modified_IntegerValued_Sample_Size}
% ====================================================
When the computational budget $b$ is small, direct flooring of the real-valued MFMC allocation may assign zero samples to high-fidelity models, leading to degenerate estimators and destroying the variance-reduction mechanism central to MFMC. To prevent this failure, we incorporate the modified rounding strategy of \cite{GrGuJuWa:2023} into our iterative allocation framework. Since successive fidelity levels sharing identical integer sample sizes do not contribute to variance reduction but still incur cost, we further refine the allocation procedure by removing such redundant levels during iteration. The resulting active subset of fidelities exhibits a strictly increasing sequence of integer sample sizes, ensuring efficient budget utilization and preservation of the hierarchical MFMC structure.







\subsection{Modified Allocation Framework}

As established in~\cite{GrGuJuWa:2023}, we are interested in the following integer-constrained optimization problem:
%
\begin{equation}\label{eq:Optimization_pb_integer}
    \begin{array}{ll}
    \min  &\sum_{k=1}^K\frac{\Delta_k}{m_k},\\
       \text{subject to} &\displaystyle\sum\limits_{k=1}^K C_km_k\le b,\\[2pt]
       &\displaystyle m_1\ge 1,\quad \displaystyle m_{k-1}-m_k\le 0, \;\; k=2\ldots,K,\\
       &m_1,\ldots, m_K\in \mathbb{N}.
       % &\alpha_2,\ldots,\alpha_K\in \mathbb{R}.
    \end{array}
\end{equation}
%
The modified allocation enforces a minimum of one sample at every fidelity level while ensuring budget adherence. Whenever the continuous solution produces $m_k^* < 1$, the integer value is rounded by ceiling, and the remaining budget is redistributed among the higher-index fidelities according to the continuous optimality condition. Let $i$ denote the first index at which the continuous solution satisfies $m_k^* \ge 1$ for all $k \ge i$. The modified real-valued allocation takes the form
%
\begin{equation}
    \label{eq:Modified_sample_size_floor_real_valued}
    x_1^* =\ldots=x_{i-1}^*= 1, \qquad x_k^* = \sqrt{\dfrac{\Delta_k}{C_k}}
        \dfrac{b - \sum_{j=1}^{i-1} C_j}
        {\sum_{j=i}^{K} \sqrt{C_j \Delta_j}},\quad  i \le k \le K,
\end{equation}
%
followed by flooring $x_k^*$ to obtain integer sample sizes. As shown in \cite{GrGuJuWa:2023}, this allocation yields the unique global minimizer of \eqref{eq:Optimization_pb_integer} under the optimal weights \eqref{eq:MFMC_weights}. The associated budget utilization and variance satisfy
%
\begin{equation}\label{eq:Modified_f_real_valued}
    \sum_{k=1}^K C_k x_k^* = b, 
    \qquad 
    f(x_k^*) = \sum_{k=1}^K \frac{\Delta_k}{x_k^*}
    = \sum_{k=1}^{i-1} \Delta_k 
    + \frac{\left(\sum_{k=i}^K \sqrt{C_k \Delta_k}\right)^2}
    {b - \sum_{k=1}^{i-1} C_k}.
\end{equation}
%



To integrate these constraints into our iterative framework, we define the recursive real-valued sequence
\begin{equation}
    \label{eq:Modified_sample_size_iterative_real_valued}
    H_1^* = \ldots = H_{i-1}^* =1, \qquad H_k^*=\sqrt{\frac{\Delta_k}{C_k}}\frac{b-\sum_{j=1}^{k-1}C_j  H_j^* }{\sum_{j=k}^{K}\sqrt{C_j\Delta_j}}, \quad i\le k\le K.
\end{equation}
which mirrors the stepwise structure of the iterative MFMC iteration but now incorporates the lower bound and budget reduction induced by the modified allocation. The following theorem shows that this iterative formulation exactly reproduces the modified allocation and thus inherits its optimal variance--cost properties.  The proof can be conducted using the same strategy mentioned in Section \ref{sec:Iterative_IntegerValued_Sample_Size}.

Under the budget admissibility condition \eqref{eq:p_bound} and standard variance assumptions on $\Delta_k$, these allocations are identical:
\[
H_k^* = x_k^*, \quad \forall k = 1,\ldots,K.
\]

\end{theorem}

\subsection{Post-processing for identical integer sample allocation}
When the computational budget $b$ is small, the iterative scheme in 
\eqref{eq:Modified_sample_size_iterative_integer_proxy}
may assign identical integer sample sizes to consecutive fidelity levels. 
Since equal allocations do not contribute to additional variance reduction 
while still incurring computational cost, such configurations are inefficient. To address this issue, we introduce a post-processing step that removes redundant fidelity levels and reallocates the budget over a reduced hierarchy. This post-processing procedure is particularly beneficial when $b$ is small relative to the number of available fidelity levels.

Let $m_k$ denote the integer allocation generated by the recursive 
strategy (either modified or unmodified). Starting from the highest fidelity level, if two consecutive levels satisfy $m_{k} = m_{k-1}$, then the $k$-th model is eliminated from the hierarchy. After eliminating such a level, the recursion is restarted from the beginning to recompute the sample allocation for the reduced model set. The parameters of the reduced hierarchy must continue to satisfy the feasibility assumptions of MFMC stated in Theorem~\ref{thm:Sample_size_real}. The correlation monotonicity condition is automatically preserved when removing an intermediate fidelity level. The only nontrivial condition that must be verified is the cost–correlation ratio inequality $\frac{\rho_{1,k-1}^2 - \rho_{1,k+1}^2}{C_{k-1}}<\frac{\rho_{1,k+1}^2 - \rho_{1,k+2}^2}{C_{k+1}}$, which ensures that the optimal allocation for the reduced hierarchy yields a 
strictly increasing sequence of sample sizes and that each retained fidelity level contributes effectively to variance reduction. If this condition is violated after elimination, the removed model must be reinstated in order to preserve the admissibility conditions of MFMC and to maintain the monotone structure required for optimal variance reduction.

The complete algorithm -- iterative allocation followed by post-process elimination -- is summarized in Algorithm~\ref{algo:Iterative_MFMC_Algo}.

\begin{algorithm}[!ht]
\caption{Iterative integer-valued sample size allocation with post-processing}
\label{algo:Iterative_MFMC_Algo}
\DontPrintSemicolon

\KwIn{Correlation coefficients  $\{\rho_{1,k}\}_{k=1}^K$, computational costs $\{C_k\}_{k=1}^K$, total budget $b$.}
\KwOut{Integer-valued sample sizes $\{\lfloor y_k \rfloor\}_{k=1}^{K_r}$.}
\hrule\vspace{1ex}

\If{$b < \sum_{j=1}^K C_j$}{
    \textbf{return} ``Insufficient budget: requires at least $\sum_{j=1}^K C_j$''.
}

Let $\boldsymbol{\rho} = [\rho_{1,1}, \ldots, \rho_{1,K}], \boldsymbol{C} = [C_{1}, \ldots, C_{K}]$. $b_1=b.$

Compute variance terms: $\Delta_k \leftarrow \rho_{1,k}^2 - \rho_{1,k+1}^2$ for $k = 1, \dots, K$ with $\rho_{1,K+1}\leftarrow 0$.

RestartFlag = true.\;

% Compute normalization constant: $S = \sum_{i=1}^K \sqrt{C_i \Delta_i}$\;
\While{RestartFlag}{

RestartFlag = false.\;
% Initialize the real-valued optimal allocation:
% \[
% M_1^* = \sqrt{\frac{\Delta_1}{C_1}} \frac{b}{ \sum_{j=1}^K \sqrt{C_j \Delta_j}},
% \qquad
% \left\lfloor M_1^* \right\rfloor = \max \left(\left\lfloor M_1^* \right\rfloor, 1\right)
% \]



\For{$k = 1$ \KwTo $K$}{
    
    Compute the current allocation:
    \[
    y_k = \sqrt{\frac{\Delta_k}{C_k}}\frac{b_k}{\sum_{j=k}^K \sqrt{C_j \Delta_j}},
    \qquad
    \left\lfloor y_k \right\rfloor = \max \left(\left\lfloor y_k \right\rfloor, 1 \right).
    \]
    % \If{$\lfloor M_k^* \rfloor < 1$}{
    %     Enforce minimum sample: $\lfloor M_k^* \rfloor = 1$.
    % }
    Update residual budget: $b_{k+1} \leftarrow b_k -  C_k \left\lfloor y_k \right\rfloor$.\;


\If{$k\ge 2$ \& $\left\lfloor y_{k-1} \right\rfloor=\left\lfloor y_k \right\rfloor$}{
    
    
    
    Compute updated variance: $\widetilde\Delta_{k-1}\leftarrow\rho_{1,k-1}^2 - \rho_{1,k+1}^2, \widetilde\Delta_k \leftarrow \rho_{1,k+1}^2-\rho_{1,k+2}^2$

    
    \If{condition $\frac{\widetilde\Delta_{k-1}}{C_{k-1}}<\frac{\widetilde\Delta_{k}}{C_{k+1}}$}{ 

    Discard model $k$, update:  $\boldsymbol{\rho} \leftarrow \boldsymbol{\rho} \setminus \{\rho_{1,k}\}$, $\boldsymbol{C} \leftarrow \boldsymbol{C} \setminus \{C_k\}$, $\Delta_k \leftarrow \rho_{1,k}^2 - \rho_{1,k+1}^2$.\;
    $K \leftarrow K - 1$.\;

    {\tt RestartFlag} = {\tt true}; \;

    \textbf{break}.\;
    % \scr{Restart iteration with updated parameters and recompute $M_k^*$ from step 5.}
    }
    
    % \Else{Retain model $k$.}
    
}
}
}
$K_r \leftarrow K$\;
\textbf{return} $\left\lfloor y_1 \right\rfloor, \dots, \left\lfloor y_{K_r} \right\rfloor$.
\end{algorithm}
